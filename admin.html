// Ye function aapke 1000 products ko ek click mein add karega function bulkUpload() { // 1. Yahan aapke 1000 products ki list aayegi const bigData = [ { name: "Product 1", price: 500, category: "Electronics", image: "link1" }, { name: "Product 2", price: 1200, category: "Fashion", image: "link2" }, // ... aise hi 1000 items yahan paste karein ]; const btn = document.getElementById("bulkBtn"); btn.innerText = "Uploading " + bigData.length + " items..."; btn.disabled = true; let count = 0; // 2. Loop jo har product ko bari-bari add karega bigData.forEach((item) => { db.ref("products").push(item) .then(() => { count++; if (count === bigData.length) { alert("Badhai ho! Saare 1000 products add ho gaye."); btn.innerText = "Upload Complete"; } }) .catch((err) => console.log("Error at item " + count + ": ", err)); }); } // --- AUTOMATIC 1000 PRODUCT GENERATOR --- function generate1000Products() { const categories = ["Electronics", "Fashion", "Home", "Grocery", "Toys"]; const names = ["Premium", "Smart", "Ultra", "Classic", "Budget", "Luxury"]; const items = ["Device", "Watch", "Shirt", "Shoes", "Bag", "Bottle", "Lamp", "Chair"]; let bulkArray = []; // 1000 baar loop chalega for (let i = 1; i <= 1000; i++) { const randomCat = categories[Math.floor(Math.random() * categories.length)]; const randomName = names[Math.floor(Math.random() * names.length)] + " " + items[Math.floor(Math.random() * items.length)] + " " + i; const randomPrice = Math.floor(Math.random() * (50000 - 100) + 100); // 100 se 50,000 ke beech price // Sample image (aap ise baad mein badal sakte hain) const imgUrl = `https://picsum.photos/seed/${i}/300/300`; bulkArray.push({ name: randomName, category: randomCat, price: randomPrice, image: imgUrl, description: "Ye ek premium quality product hai jo Amazan shop par milta hai." }); } // Firebase mein upload karna const btn = document.getElementById("bulkBtn"); btn.innerText = "Processing 1000 Items..."; btn.disabled = true; // Firebase database mein 'products' node par pura array bhej rahe hain db.ref("products").set(bulkArray) // 'set' ka matlab hai purana data hat jayega aur naya 1000 aa jayega .then(() => { alert("Badhai ho Vinod! 1000 products add ho gaye hain."); btn.innerText = "1000 Products Added!"; }) .catch((err) => { alert("Kuch galti hui: " + err.message); btn.disabled = false; }); }

function generate1000Products() { const categories = ["Electronics", "Fashion", "Home", "Grocery"]; const items = ["Pro", "Max", "Smart", "Ultra", "Classic", "Mini"]; const products = ["Phone", "Watch", "Shirt", "Shoes", "Laptop", "Bag", "Bottle", "T-Shirt"]; let bulkData = {}; // Hum ek object banayenge const btn = document.getElementById("bulkBtn"); btn.innerText = "Generating 1000 Products... Please Wait"; btn.disabled = true; // 1 to 1000 ka loop for (let i = 1; i <= 1000; i++) { const randomCat = categories[Math.floor(Math.random() * categories.length)]; const randomName = items[Math.floor(Math.random() * items.length)] + " " + products[Math.floor(Math.random() * products.length)] + " " + i; const randomPrice = Math.floor(Math.random() * 50000) + 199; // Price 199 se 50,000 ke beech // Har product ki apni unique photo (Lorem Picsum se) const randomImg = `https://picsum.photos/seed/${i + 100}/300/300`; // Unique ID ke saath product data const newPostKey = db.ref().child('products').push().key; bulkData['/products/' + newPostKey] = { name: randomName, category: randomCat, price: randomPrice, image: randomImg, id: newPostKey }; } // Firebase database mein ek saath update karna (Faster than push) db.ref().update(bulkData) .then(() => { alert("Badhai ho Vinod! Aapki shop mein 1000 products add ho gaye."); btn.innerText = "1000 Products Added Successfully!"; }) .catch((error) => { alert("Galti hui: " + error.message); btn.disabled = false; }); }
